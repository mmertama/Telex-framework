cmake_minimum_required (VERSION 3.16)

set(NAME audio)
project (${NAME}test)

set(CMAKE_CXX_STANDARD 17)

include_directories(
     ../../gempyrelib/include
    include
)

add_executable(${PROJECT_NAME}
    src/main.cpp
    gui/${NAME}.html
    gui/gempyre.ogg
    include/audio_resource.h
